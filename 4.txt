b) Important Docker commands (container/content management)

Write like this in exam:

# Check Docker version
docker --version

# List images
docker images

# Pull an image
docker pull nginx

# Run a container
docker run -d --name myweb -p 8080:80 nginx

# List running containers
docker ps

# List all containers
docker ps -a

# See logs of a container
docker logs myweb

# Open shell inside container
docker exec -it myweb /bin/bash

# Stop a container
docker stop myweb

# Start a container
docker start myweb

# Remove a container
docker rm myweb

# Remove an image
docker rmi nginx

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.edge.EdgeDriver;

public class TestEdge {
    public static void main(String[] args) {

        // Set path of Edge Driver
        System.setProperty("webdriver.edge.driver", "C:\\drivers\\msedgedriver.exe");

        // Open Edge Browser
        WebDriver driver = new EdgeDriver();

        // Open Google
        driver.get("https://www.google.com");
        System.out.println("Google Title: " + driver.getTitle());

        // Open MGIT Website
        driver.get("https://mgit.ac.in");
        System.out.println("MGIT Title: " + driver.getTitle());

        // Close Browser
        driver.quit();
    }
}
